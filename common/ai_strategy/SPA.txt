	allowed = {
		OR = {
			original_tag = SPR

		}
	}
	enable = {
		SPR = { has_war_with = SPA }
	}
	abort = {
		OR = {
			AND = {
				original_tag = SPR
				SPA = { exists = no }
			}
			AND = {
				original_tag = SPA
				SPR = { exists = no }
			}
		}
	}

		type = land_xp_spend_priority 	#陆军经验花费优先级
		id = division_template 			#编辑陆军编制
		value = -9999					#千万不要花在这里
	}
}
