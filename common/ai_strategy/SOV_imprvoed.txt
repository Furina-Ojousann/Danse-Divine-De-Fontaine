SOV_europe_is_dangerous = {
	allowed = {
		original_tag = SOV 
	}
	enable = {
		OR = {
			GER = {
				AND = {
					exists = yes 
					not = { is_in_faction_with = SOV }
					has_offensive_war = yes 
				}
			}
			ENG = {
				AND = {
					exists = yes 
					not = { is_in_faction_with = SOV }
					OR = {
						has_war_with = SOV 
						focus_progress = {
							focus = ENG_war_with_ussr 
							progress > 0 
						}
					}
				}
			}
			USA = {
				AND = {
					exists = yes 
					not = { is_in_faction_with = SOV }
					has_war_with = SOV  
				}
			}
			FRA = {
				AND = {
					exists = yes 
					not = { is_in_faction_with = SOV }
					has_war_with = SOV  
				}
			}
		}
	}
	abort = {
		any_country = {
			or = {
				tag = USA
				tag = FRA
				tag = ENG
				tag = GER
				tag = ITA 
			}
			is_faction_leader = yes 
			has_capitulated = yes 
		}
		has_war = no 
	}
	ai_strategy = {
		type = area_priority
		id = europe 
		value = 999 
	}
	ai_strategy = {
		type = area_priority
		id = scandinavia
		value = 35
	}
}
