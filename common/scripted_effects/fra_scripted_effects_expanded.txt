R_Fontaine_Start_Vote_Animation = {

    # 1) 存档：原本各党席位（如果你需要动画结束后恢复）
    set_variable = { r_fontaine_saved_comm_seats = R_Fontaine_Communist_seats }
    set_variable = { r_fontaine_saved_soc_seats  = R_Fontaine_Socialist_seats }
    set_variable = { r_fontaine_saved_lib_seats  = R_Fontaine_Liberalist_seats }
    set_variable = { r_fontaine_saved_agr_seats  = R_Fontaine_Agrarianist_seats }
    set_variable = { r_fontaine_saved_tory_seats = R_Fontaine_Toryism_seats }
    set_variable = { r_fontaine_saved_neu_seats  = R_Fontaine_Neutralist_seats }

    # 2) 计算：本次法案支持席位（复用你已有的方法）
    R_Fontaine_Bill_support = yes

    # 这里把“支持总席位”收拢成一个变量（你可按自己阵营定义支持方/反对方）
    # 示例：把各党支持席位加总为 r_fontaine_vote_support_total
    set_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_AD }
    add_to_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_FR }
    add_to_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_CF }
    add_to_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_PF }
    add_to_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_PTF }
    add_to_temp_variable = { tmp_support_total = r_fontaine_bill_support_seats_independent }
    round_temp_variable = var:tmp_support_total

    set_variable = { r_fontaine_vote_support_total = tmp_support_total }

    # 反对总席位 = 600 - 支持总席位
    set_temp_variable = { tmp_oppose_total = 600 }
    subtract_from_temp_variable = { tmp_oppose_total = tmp_support_total }
    round_temp_variable = tmp_oppose_total
    set_variable = { r_fontaine_vote_oppose_total = tmp_oppose_total }

    # 3) UI 状态：标记“正在表决”（你 GUI 的 visible/trigger 会读这个 flag）【:contentReference[oaicite:4]{index=4}】
    set_country_flag = r_fontaine_any_bill_ongoing

    # 4) 清屏：全部置中立（frame=1）
    clear_array = r_seat_frame
    resize_array = { array = r_seat_frame size = 600 value = 1 }

    # 5) 初始化步数并触发第一帧
    set_variable = { r_fontaine_vote_step = 0 }
    country_event = { id = R_fontaine_parliament.1001 hours = 1 }
	log = "执行一次议会动画"
}

R_Fontaine_Vote_Animation_Refresh_Frames = {

    # step: 1~10
    set_temp_variable = { tmp_step = r_fontaine_vote_step }

    # revealed_support = round( support_total * step / 10 )
    set_temp_variable = { tmp_support = r_fontaine_vote_support_total }
    multiply_temp_variable = { tmp_support = tmp_step }
    divide_temp_variable   = { tmp_support = 10 }
    round_temp_variable    = tmp_support

    # revealed_oppose = round( oppose_total * step / 10 )
    set_temp_variable = { tmp_oppose = r_fontaine_vote_oppose_total }
    multiply_temp_variable = { tmp_oppose = tmp_step }
    divide_temp_variable   = { tmp_oppose = 10 }
    round_temp_variable    = tmp_oppose

    # sum = revealed_oppose + revealed_support
    set_temp_variable = { tmp_sum = tmp_oppose }
    add_to_temp_variable = { tmp_sum = tmp_support }

    # 重建 r_seat_frame：
    # 红：反对（用 Communist 的帧值 6）【:contentReference[oaicite:6]{index=6}】
    # 蓝：支持（用 Liberalist 的帧值 4）【:contentReference[oaicite:7]{index=7}】
    # 灰：未展示/中立（帧值 1）【:contentReference[oaicite:8]{index=8}】
    clear_array = r_seat_frame
    resize_array = { array = r_seat_frame size = tmp_oppose value = 6 }
    resize_array = { array = r_seat_frame size = tmp_sum   	value = 4 }
    resize_array = { array = r_seat_frame size = 600       	value = 1 }
}
