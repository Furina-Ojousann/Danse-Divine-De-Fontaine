						modifier = FRA_the_corruption

			}

		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = GER 
						has_government = neutrality 
					}
					FROM = {
						original_tag = GER
						has_government = fascism 
					}
				}
				retire_character = GER_rudolf_hess 
				add_advisor_role = {
					character = GER_ludwig_beck
					activate = yes 
					advisor = {
						slot = second_chief
						traits = {
							rearmament_proponent
						}
					}
				}
			}
		}
	}
	on_leave_faction = {
		effect = {
			if = {
			limit = {
				has_country_flag = comintern_partner 
			}
			clr_country_flag = comintern_partner
		}
		if = {
			limit = {
				is_faction_leader = yes

			}
			clr_country_flag = comintern_leader 
		}
		}
		
	}

			}
		}
	}
