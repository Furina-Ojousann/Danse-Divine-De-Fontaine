# 1900: ENG 隐藏结算事件（等两边都表态后分支）
# 规则：
# - BEL同意 & FRA同意 -> ENG20 -> FRA21
# - BEL拒绝 & FRA同意 -> ENG & FRA 同时 22
# - BEL同意 & FRA拒绝 -> ENG18
# - BEL拒绝 & FRA拒绝 -> ENG23 -> FRA13(只剩拒绝选项)
# ============================================================
country_event = {
    id = R_Fontaine_diplomacy.1900
    hidden = yes
    is_triggered_only = yes

    immediate = {
        # 两边是否都已表态？
        if = {
            limit = {
                OR = { BEL = { has_country_flag = r_fon_diplo_bel_accept } BEL = { has_country_flag = r_fon_diplo_bel_refuse } }
                OR = { FRA = { has_country_flag = r_fon_diplo_fra_accept } FRA = { has_country_flag = r_fon_diplo_fra_refuse } }
            }

            # 结算分支
            if = {
                limit = {
                    BEL = { has_country_flag = r_fon_diplo_bel_accept }
                    FRA = { has_country_flag = r_fon_diplo_fra_accept }
                }
                country_event = { id = R_Fontaine_diplomacy.20 hours = 12 }
            }
            else_if = {
                limit = {
                    BEL = { has_country_flag = r_fon_diplo_bel_refuse }
                    FRA = { has_country_flag = r_fon_diplo_fra_accept }
                }
                country_event = { id = R_Fontaine_diplomacy.22 hours = 12 }
                FRA = { country_event = { id = R_Fontaine_diplomacy.22 hours = 12 } }
            }
            else_if = {
                limit = {
                    BEL = { has_country_flag = r_fon_diplo_bel_accept }
                    FRA = { has_country_flag = r_fon_diplo_fra_refuse }
                }
                country_event = { id = R_Fontaine_diplomacy.18 hours = 12 }
            }
            else = { # 双方都拒绝
                country_event = { id = R_Fontaine_diplomacy.23 hours = 12 }
            }

            # 清理回应，避免重复结算
            BEL = {
                clr_country_flag = r_fon_diplo_bel_accept
                clr_country_flag = r_fon_diplo_bel_refuse
            }
            FRA = {
                clr_country_flag = r_fon_diplo_fra_accept
                clr_country_flag = r_fon_diplo_fra_refuse
            }
        }
    }

    option = { name = OK }
}
